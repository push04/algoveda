from algoveda.core import Strategy, Backtest
from datetime import timedelta
# Define high-frequency mean reversion parameters
class MeanReversionStrategy(Strategy):
def initialize(self):
self.lookback = 20
self.std_dev = 2.0
self.asset = "BTC-USDT"
def on_data(self, slice):
upper, mid, lower = self.indicators.bollinger(self.asset, self.lookback)
if slice[self.asset].price < lower:
